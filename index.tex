    <script>
        const colors = ["#ff0a54", "#ff477e", "#ff85a1", "#fbb1bd", "#ffd60a", "#00f5d4", "#70e000", "#ffffff", "#4cc9f0"];

        function createConfetti(isBurst = false) {
            const confetti = document.createElement("div");
            confetti.classList.add("confetti");
            
            // Random styling
            const color = colors[Math.floor(Math.random() * colors.length)];
            confetti.style.background = color;
            confetti.style.left = Math.random() * 100 + "vw";
            
            // If it's a burst, start from the middle/bottom area
            if(isBurst) {
                confetti.style.top = "50vh";
                confetti.style.left = "50vw";
                confetti.style.width = "12px";
                confetti.style.height = "12px";
            } else {
                confetti.style.top = "-5vh";
            }

            const duration = (Math.random() * 3 + 3);
            confetti.style.animation = `fall ${duration}s linear forwards`;
            
            // Add variety in shapes
            if(Math.random() > 0.5) confetti.style.borderRadius = "50%";

            document.body.appendChild(confetti);

            setTimeout(() => {
                confetti.remove();
            }, 5000);
        }

        function showWish() {
            const msg = document.getElementById("message");
            msg.style.display = "block";
            
            // Trigger massive burst
            for(let i = 0; i < 100; i++) {
                setTimeout(() => {
                    createConfetti(true);
                }, i * 5);
            }
            
            // Change button text
            const btn = document.querySelector('button');
            btn.innerHTML = "Enjoy Your Day! 🥳";
            btn.style.background = "#fff8b0";
        }

        // Background steady rain of confetti
        setInterval(createConfetti, 300);

        // Add some random floating balloons
        function createBalloon() {
            const balloon = document.createElement("div");
            balloon.className = "balloon";
            const emojis = ['🎈', '✨', '💖', '🎂', '⭐'];
            balloon.innerHTML = emojis[Math.floor(Math.random() * emojis.length)];
            balloon.style.fontSize = Math.random() * 30 + 20 + "px";
            balloon.style.left = Math.random() * 100 + "vw";
            balloon.style.animationDuration = Math.random() * 5 + 5 + "s";
            document.body.appendChild(balloon);
            
            setTimeout(() => {
                balloon.remove();
            }, 10000);
        }

        setInterval(createBalloon, 2000);
    </script>
